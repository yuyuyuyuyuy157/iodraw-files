```mermaid
graph TD
   start=>start: 开始
input=>input: 初始化图形界面\n初始化多项式A和B
menu=>condition: 主菜单选择
create=>operation: 输入多项式A和B
add=>operation: A + B
sub=>operation: A - B
mul=>operation: A * B
eval=>condition: 求值菜单选择
eval_A=>operation: 计算 A(x)
eval_B=>operation: 计算 B(x)
eval_add=>operation: 计算 (A+B)(x)
eval_sub=>operation: 计算 (A-B)(x)
eval_mul=>operation: 计算 (A*B)(x)
back=>operation: 返回上一级菜单
exit=>end: 退出程序

start->input->menu

menu(yes)->create->menu
menu(no)->condition->exit

menu=>condition
menu(yes)->create
menu(no)->add
menu(no)->sub
menu(no)->mul
menu(no)->eval
menu(no)->exit

eval(yes)->eval_A->back->menu
eval(no)->eval_B->back->menu
eval(no)->eval_add->back->menu
eval(no)->eval_sub->back->menu
eval(no)->eval_mul->back->menu
```
